class PropertyFilter:

        self,
        ...,
        published_after: Optional[datetime] = None,
        published_before: Optional[datetime] = None,
        keyword: Optional[str] = None,
        sort_by: Optional[str] = "price",  # price, area, date
        sort_order: str = "asc"

    def filter(self, properties: List[PropertyCreate]) -> List[PropertyCreate]:
        filtered = []
        
        for prop in properties:
            # ... existing filters
            
            # Date filtering
            if published_after and prop.created_at < published_after:
                continue
            
            # Keyword search
            if keyword and keyword.lower() not in prop.title.lower():
                continue
            
            filtered.append(prop)
        
        # Sorting
        if sort_by == "price":
            key_func = lambda x: x.price
        elif sort_by == "area":
            key_func = lambda x: x.area or 0
        else:
            key_func = lambda x: x.created_at
        
        return sorted(
            filtered,
            key=key_func,
            reverse=(sort_order == "desc")
